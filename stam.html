name: Deploy
on:
push:
branches:
– main
#- repository_dispatch
jobs:
deploy:
runs-on: ubuntu-latest
timeout-minutes: 10
steps:
– uses: actions/checkout@v2
– uses: actions/setup-node@v1
with:
node-version: 12
– run: yarn install
– run: yarn build
– name: Publish
uses: cloudflare/wrangler-action@1.2.0
with:
apiToken: ${{ secrets.CF_API_TOKEN }}
env:
CF_ACCOUNT_ID: ${{ secrets.CF_ACCOUNT_ID }}
IS_WORKER: true
